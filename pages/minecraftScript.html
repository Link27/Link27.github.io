#!/bin/bash

echo "****************************"
echo "Backup Minecraft Server"
echo "****************************"

BackupWorldLoc="*insert backup world location*"
CurrentWorldLoc="*insert current world directory location*"

if [ -d "${BackupWorldLoc}Backup1" ]; then
 echo "Backup1 Does Exist"
 if [ -d "${BackupWorldLoc}Backup2" ]; then
  echo "Backup2 Does Exist"
  if [ -d "${BackupWorldLoc}Backup3" ]; then
   echo "Backup3 Does Exist"
   if [ -d "${BackupWorldLoc}Backup4" ]; then
    echo "Backup4 Does Exist"
    if [ -d "${BackupWorldLoc}Backup5" ]; then
     echo "Backup5 Does Exist"
     if [ -d "${BackupWorldLoc}Backup6" ]; then
      echo "Backup6 Does Exist"
      if [ -d "${BackupWorldLoc}Backup7" ]; then
       echo "Backup7 Does Exist"
       if [ -d "${BackupWorldLoc}Backup8" ]; then
        echo "Backup8 Does Exist"
        rm -r "${BackupWorldLoc}Backup1/"
        echo "Backup1 Has Been Removed"
        cp -r "${BackupWorldLoc}Backup2/." "${BackupWorldLoc}Backup1/"
        rm -r "${BackupWorldLoc}Backup2/"
        echo "Backup2 Has Been Removed"
        cp -r "${BackupWorldLoc}Backup3/." "${BackupWorldLoc}Backup2/"
        rm -r "${BackupWorldLoc}Backup3/"
        echo "Backup3 Has Been Removed"
        cp -r "${BackupWorldLoc}Backup4/." "${BackupWorldLoc}Backup3/"
        rm -r "${BackupWorldLoc}Backup4/"
        echo "Backup4 Has Been Removed"
        cp -r "${BackupWorldLoc}Backup5/." "${BackupWorldLoc}Backup4/"
        rm -r "${BackupWorldLoc}Backup5/"
        echo "Backup5 Has Been Removed"
        cp -r "${BackupWorldLoc}Backup6/." "${BackupWorldLoc}Backup5/"
        rm -r "${BackupWorldLoc}Backup6/"
        echo "Backup6 Has Been Removed"
        cp -r "${BackupWorldLoc}Backup7/." "${BackupWorldLoc}Backup6/"
        rm -r "${BackupWorldLoc}Backup7/"
        echo "Backup7 Has Been Removed"
        cp -r "${BackupWorldLoc}Backup8/." "${BackupWorldLoc}Backup7/"
        rm -r "${BackupWorldLoc}Backup8/"
        echo "Backup8 Has Been Removed"
        cp -r "${CurrentWorldLoc}world/." "${BackupWorldLoc}Backup8/"
        echo "Backup8 Has Been Created"
       else
        echo "Backup8 Does Not Exist"
        cp -r "${CurrentWorldLoc}world/." "${BackupWorldLoc}Backup8/"
        echo "Backup8 Has Been Created"
       fi
      else
       echo "Backup7 Does Not Exist"
       cp -r "${CurrentWorldLoc}world/." "${BackupWorldLoc}Backup7/"
       echo "Backup7 Has Been Created"
      fi
     else
      echo "Backup6 Does Not Exist"
      cp -r "${CurrentWorldLoc}world/." "${BackupWorldLoc}Backup6/"
      echo "Backup6 Has Been Created"
     fi
    else
     echo "Backup5 Does Not Exist"
     cp -r "${CurrentWorldLoc}world/." "${BackupWorldLoc}Backup5/"
     echo "Backup5 Has Been Created"
    fi
   else
    echo "Backup4 Does Not Exist"
    cp -r "${CurrentWorldLoc}world/." "${BackupWorldLoc}Backup4/"
    echo "Backup4 Has Been Created"
   fi
  else
   echo "Backup3 Does Not Exist"
   cp -r "${CurrentWorldLoc}world/." "${BackupWorldLoc}Backup3/"
   echo "Backup3 Has Been Created"
  fi
 else
  echo "Backup2 Does Not Exist"
  cp -r "${CurrentWorldLoc}world/." "${BackupWorldLoc}Backup2/"
  echo "Backup2 Has Been Created"
 fi
else
 echo "Backup1 Does Not Exist"
 cp -r "${CurrentWorldLoc}world/." "${BackupWorldLoc}Backup1/"
 echo "Backup1 Has Been Created"
fi
